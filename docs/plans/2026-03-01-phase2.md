### Task 1: Install next-intl and create routing config

**Files:**

**Step 1: Install dependency**

```bash
npm install next-intl
```

Expected output: `added N packages`

**Step 2: Create `i18n/routing.ts`**

```ts
import { defineRouting } from "next-intl/routing";

export const routing = defineRouting({
  locales: ["de", "fr", "it", "en"],
  defaultLocale: "de",
  localePrefix: "as-needed", // DE has no prefix; FR/IT/EN get /fr/ /it/ /en/
});
```

**Step 3: Create `i18n/request.ts`**

```ts
import { getRequestConfig } from "next-intl/server";
import { routing } from "./routing";

export default getRequestConfig(async ({ requestLocale }) => {
  let locale = await requestLocale;
  if (!locale || !routing.locales.includes(locale as "de" | "fr" | "it" | "en")) {
    locale = routing.defaultLocale;
  }
  return {
    locale,
    messages: (await import(`../messages/${locale}.json`)).default,
  };
});
```

**Step 4: Create `middleware.ts` in project root**

```ts
import createMiddleware from "next-intl/middleware";
import { routing } from "./i18n/routing";

export default createMiddleware(routing);

export const config = {
  matcher: [
    "/((?!api|_next/static|_next/image|favicon.ico|robots.txt|sitemap.xml|.*\\.(?:svg|png|jpg|jpeg|gif|webp|ico|json)).*)",
  ],
};
```

**Step 5: Update `next.config.ts`**

```ts
import type { NextConfig } from "next";
import createNextIntlPlugin from "next-intl/plugin";

const withNextIntl = createNextIntlPlugin("./i18n/request.ts");

      {
        source: "/informationen",
        destination: "/ueber-uns",
        permanent: true,
      },
    ];
  },
};

export default withNextIntl(nextConfig);
```

```tsx
"use client";

import { useLocale } from "next-intl";
import { usePathname, useRouter } from "next/navigation";
import { useState, useRef, useEffect } from "react";
import { Globe } from "lucide-react";

const LOCALES = [
  { code: "de", label: "DE" },
  { code: "fr", label: "FR" },
  { code: "it", label: "IT" },
  { code: "en", label: "EN" },
] as const;

// Build the locale-prefixed path for switching
function buildLocalePath(currentPath: string, currentLocale: string, targetLocale: string): string {
  const prefixed = ["fr", "it", "en"];
  // Strip current locale prefix if present
  let path = currentPath;
  if (prefixed.includes(currentLocale)) {
    path = path.replace(new RegExp(`^/${currentLocale}`), "") || "/";
  }
  // Add target locale prefix
  if (prefixed.includes(targetLocale)) {
    return `/${targetLocale}${path === "/" ? "" : path}`;
  }
  return path || "/";
}

export default function LocaleSwitcher() {
  const locale = useLocale();
  const pathname = usePathname();
  const router = useRouter();
  const [open, setOpen] = useState(false);
  const ref = useRef<HTMLDivElement>(null);

  useEffect(() => {
    function handleClickOutside(e: MouseEvent) {
      if (ref.current && !ref.current.contains(e.target as Node)) setOpen(false);
    }
    document.addEventListener("mousedown", handleClickOutside);
    return () => document.removeEventListener("mousedown", handleClickOutside);
  }, []);

  return (
    <div ref={ref} className="relative">
      <button
        type="button"
        onClick={() => setOpen((v) => !v)}
        className="flex items-center gap-1 px-2 py-1.5 rounded-lg text-xs font-semibold text-[#374151] hover:text-[var(--ct-cyan)] transition-colors"
        aria-label="Sprache wählen"
      >
        <Globe size={14} />
        {locale.toUpperCase()}
      </button>
      {open && (
        <div className="absolute right-0 top-full mt-1 bg-white border border-[#e5e7eb] rounded-xl shadow-lg py-1 min-w-[80px] z-50">
          {LOCALES.map(({ code, label }) => (
            <button
              key={code}
              type="button"
              onClick={() => {
                setOpen(false);
                const target = buildLocalePath(pathname, locale, code);
                router.push(target);
              }}
              className={`w-full text-left px-3 py-2 text-xs font-semibold transition-colors ${
                code === locale
                  ? "text-[var(--ct-cyan)]"
                  : "text-[#374151] hover:text-[var(--ct-cyan)] hover:bg-[var(--ct-light)]"
              }`}
            >
              {label}
            </button>
          ))}
        </div>
      )}
    </div>
  );
}
```

The compare state lives in the URL query param `?compare=1,3,5`. This helper parses/serializes it.

```ts
export const MAX_COMPARE = 3;

export function getCompareIds(searchParams: URLSearchParams): number[] {
  const raw = searchParams.get("compare");
  if (!raw) return [];
  return raw.split(",").map(Number).filter(Boolean);
}

export function buildCompareUrl(ids: number[], baseSearch: string): string {
  const params = new URLSearchParams(baseSearch);
  if (ids.length === 0) {
    params.delete("compare");
  } else {
    params.set("compare", ids.join(","));
  }
  const str = params.toString();
  return str ? `?${str}` : "";
}
```

```ts
import { NextRequest, NextResponse } from "next/server";
import { kv } from "@vercel/kv";
import { Resend } from "resend";
import { fetchVehicles } from "@/lib/vehicles";
import { formatCHF } from "@/lib/utils";

const resend = new Resend(process.env.RESEND_API_KEY);

// Protect cron with Vercel's built-in CRON_SECRET header
function authorized(req: NextRequest): boolean {
  const auth = req.headers.get("authorization");
  return auth === `Bearer ${process.env.CRON_SECRET}`;
}

export async function GET(req: NextRequest) {
  if (!authorized(req)) {
    return NextResponse.json({ error: "Unauthorized" }, { status: 401 });
  }

  try {
    const alertIds = await kv.smembers("alerts") as string[];
    if (!alertIds.length) return NextResponse.json({ checked: 0 });

    const vehicles = await fetchVehicles();
    let matched = 0;

    for (const id of alertIds) {
      const alert = await kv.get<{
        email: string;
        filters: Record<string, string>;
        createdAt: number;
      }>(`alert:${id}`);
      if (!alert) continue;

      // Simple match: find vehicles newer than alert creation that match make/model filters
      const matches = vehicles.filter((v) => {
        const f = alert.filters;
        if (f.make && v.make.toLowerCase() !== f.make.toLowerCase()) return false;
        if (f.model && !v.model.toLowerCase().includes(f.model.toLowerCase())) return false;
        if (f.priceMax && v.price > Number(f.priceMax)) return false;
        return true;
      });

      if (matches.length > 0) {
        const top = matches[0];
        await resend.emails.send({
          from: process.env.RESEND_FROM ?? "noreply@cartrade24.ch",
          to: alert.email,
          subject: `Neues Fahrzeug: ${top.make} ${top.model} – CHF ${formatCHF(top.price)}`,
          html: `
            <p>Guten Tag</p>
            <p>Ein passendes Fahrzeug ist jetzt verfügbar:</p>
            <p><strong>${top.make} ${top.model}${top.variant ? " " + top.variant : ""}</strong><br/>
            CHF ${formatCHF(top.price)} · ${top.year} · ${formatCHF(top.mileage)} km</p>
            <p><a href="https://cartrade24.ch/autos/${top.id}">Fahrzeug ansehen →</a></p>
            <p>Mit freundlichen Grüssen<br/>Car Trade24 GmbH</p>
          `,
        });
        matched++;
      }
    }

    return NextResponse.json({ checked: alertIds.length, matched });
  } catch (err) {
    console.error("price-check cron error", err);
    return NextResponse.json({ error: "Interner Fehler" }, { status: 500 });
  }
}
```

```ts
import { NextRequest, NextResponse } from "next/server";
import Stripe from "stripe";
import { kv } from "@vercel/kv";
import { Resend } from "resend";
import { fetchVehicles } from "@/lib/vehicles";
import { formatCHF } from "@/lib/utils";

const stripe = new Stripe(process.env.STRIPE_SECRET_KEY!);
const resend = new Resend(process.env.RESEND_API_KEY);
const RESERVATION_HOURS = Number(process.env.RESERVATION_HOURS ?? 48);

// Next.js App Router: disable body parsing for raw body access
export const config = { api: { bodyParser: false } };

export async function POST(req: NextRequest) {
  const sig = req.headers.get("stripe-signature");
  const rawBody = await req.text();

  let event: Stripe.Event;
  try {
    event = stripe.webhooks.constructEvent(rawBody, sig!, process.env.STRIPE_WEBHOOK_SECRET!);
  } catch (err) {
    console.error("Webhook signature error", err);
    return NextResponse.json({ error: "Invalid signature" }, { status: 400 });
  }

  if (event.type === "checkout.session.completed") {
    const session = event.data.object as Stripe.Checkout.Session;
    const vehicleId = session.metadata?.vehicleId;
    const customerEmail = session.customer_details?.email;
    const vehicleLabel = session.metadata?.vehicleLabel ?? "Fahrzeug";

    if (vehicleId) {
      // Mark vehicle as reserved for RESERVATION_HOURS
      await kv.set(
        `reserved:${vehicleId}`,
        { until: Date.now() + RESERVATION_HOURS * 3600 * 1000, sessionId: session.id },
        { ex: RESERVATION_HOURS * 3600 }
      );

      // Notify customer
      if (customerEmail) {
        const vehicles = await fetchVehicles();
        const vehicle = vehicles.find((v) => v.id === Number(vehicleId));
        await resend.emails.send({
          from: process.env.RESEND_FROM ?? "noreply@cartrade24.ch",
          to: customerEmail,
          subject: `Reservierungsbestätigung – ${vehicleLabel}`,
          html: `
            <p>Guten Tag</p>
            <p>Ihre Reservierung für <strong>${vehicleLabel}</strong>${vehicle ? ` (CHF ${formatCHF(vehicle.price)})` : ""} wurde bestätigt.</p>
            <p>Das Fahrzeug ist für <strong>${RESERVATION_HOURS} Stunden</strong> für Sie reserviert.</p>
            <p>Unser Team meldet sich in Kürze bei Ihnen.</p>
            <p>Mit freundlichen Grüssen<br/>Car Trade24 GmbH<br/>+41 56 618 55 44</p>
          `,
        });
      }

      // Notify dealer
      await resend.emails.send({
        from: process.env.RESEND_FROM ?? "noreply@cartrade24.ch",
        to: "info@cartrade24.ch",
        subject: `Neue Reservierung: ${vehicleLabel}`,
        html: `
          <p>Neues Reservierungs-Checkout abgeschlossen.</p>
          <p>Fahrzeug: <strong>${vehicleLabel}</strong> (ID: ${vehicleId})</p>
          <p>Kunde: ${customerEmail ?? "unbekannt"}</p>
          <p>Stripe Session: ${session.id}</p>
        `,
      });
    }
  }

  return NextResponse.json({ received: true });
}
```

```ts
export interface Valuation {
  min: number;
  max: number;
  currency: "CHF";
}

export async function getValuation(
  plate: string,
  km: number,
  condition: "Sehr gut" | "Gut" | "Gebraucht"
): Promise<Valuation> {
  if (!process.env.EUROTAX_API_KEY) {
    // Demo mode: plausible estimate based on km and condition
    const conditionFactor = condition === "Sehr gut" ? 1.1 : condition === "Gut" ? 1.0 : 0.85;
    const base = Math.max(4000, (35000 - km * 0.2) * conditionFactor);
    return {
      min: Math.round(base * 0.85 / 100) * 100,
      max: Math.round(base * 1.15 / 100) * 100,
      currency: "CHF",
    };
  }

  // Real Eurotax API call
  const url = `${process.env.EUROTAX_API_URL}/valuate`;
  const res = await fetch(url, {
    method: "POST",
    headers: {
      "Authorization": `Bearer ${process.env.EUROTAX_API_KEY}`,
      "Content-Type": "application/json",
    },
    body: JSON.stringify({ plate, km, condition }),
  });
  if (!res.ok) throw new Error(`Eurotax API error: ${res.status}`);
  const data = await res.json();
  return { min: data.min, max: data.max, currency: "CHF" };
}
```

```ts
import { NextRequest, NextResponse } from "next/server";
import { getValuation } from "@/lib/eurotax";

export async function POST(req: NextRequest) {
  try {
    const { plate, km, condition } = await req.json();
    if (!plate || !km || !condition) {
      return NextResponse.json({ error: "Fehlende Parameter" }, { status: 400 });
```

```ts
import { NextRequest, NextResponse } from "next/server";
import Stripe from "stripe";

const stripe = new Stripe(process.env.STRIPE_SECRET_KEY!);
const TRADEIN_FEE = Number(process.env.TRADEIN_FEE_CHF ?? 20);

export async function POST(req: NextRequest) {
  try {
    const { plate, km, condition, locale = "de" } = await req.json();
    const baseUrl = process.env.NEXT_PUBLIC_BASE_URL ?? "https://cartrade24.ch";
    const localePrefix = ["fr", "it", "en"].includes(locale) ? `/${locale}` : "";

    const session = await stripe.checkout.sessions.create({
      mode: "payment",
      payment_method_types: ["card"],
      line_items: [
        {
          quantity: 1,
          price_data: {
            currency: "chf",
            unit_amount: TRADEIN_FEE * 100,
            product_data: {
              name: "Verbindliche Fahrzeugbewertung",
              description: `Kontrollschild: ${plate} · ${km} km · Zustand: ${condition}`,
            },
          },
        },
      ],
      metadata: { plate, km: String(km), condition, locale },
      success_url: `${baseUrl}${localePrefix}/inzahlungnahme/bestaetigung?session_id={CHECKOUT_SESSION_ID}`,
      cancel_url: `${baseUrl}${localePrefix}/inzahlungnahme`,
    });

    return NextResponse.json({ url: session.url });
  } catch (err) {

    return NextResponse.json({ error: "Interner Fehler" }, { status: 500 });
  }
}
```
